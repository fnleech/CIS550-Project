CREATE TABLE Olympics (
 Year VARCHAR(12), 
 Opening_ceremony VARCHAR(25), 
 Closing_ceremony VARCHAR(25), 
 Cost VARCHAR(25), 
 Primary Key (Year)
);

CREATE TABLE Participates (
 Year INTEGER,
 CID VARCHAR(12), 
 Role VARCHAR(25), 
 Primary Key (Year, CID, Role),
 Foreign Key (Year) References Olympics(Year),
 Foreign Key (CID) References Country(CID),
 CHECK (role IN ('host','participate','both', 'none'))
);

CREATE TABLE Country (
 CID VARCHAR(12), 
 Name VARCHAR(80), 
 Continent VARCHAR(50), 
 Population VARCHAR(50), 
 GDP VARCHAR(25),  
 PM10 VARCHAR(20), 
 PM25 VARCHAR(20),
 Primary Key (CID)
);

CREATE TABLE Athlete (
 AID VARCHAR(25), 
 FullName VARCHAR(120), 
 DOB VARCHAR(25), 
 Gender VARCHAR(50), 
 Height VARCHAR(50), 
 Weight VARCHAR(50), 
 Primary Key (AID, DOB)
);

CREATE TABLE From ( 
 AID VARCHAR(25),
 CName VARCHAR(12),  
 CID VARCHAR(12), 
 Primary Key (CID, AID)
 Foreign Key (CID) References Country(CID),
 Foreign Key (AID) References Athlete(AID)
);

CREATE TABLE Sport ( 
 Discipline VARCHAR(50), 
 Event VARCHAR(100), 
 Primary Key (Discipline, Event)
);

CREATE TABLE Compete_in (
 AID VARCHAR(25),
 OlympicNo VARCHAR(25), 
 Location VARCHAR(25), 
 Discipline VARCHAR(50), 
 Event VARCHAR(100), 
 Medal VARCHAR(25), 
 Primary Key (AID, Discipline, Event, Medal),
 Foreign Key (Discipline, Event) References Sport(Discipline, Event),
 Foreign Key (AID) References Athlete(AID),
 CHECK (Medal IN ('Gold','Silver','Bronze', 'None')
);
